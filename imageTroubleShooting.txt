# This is how i describe to make the video from png files.
# you must already be in the directory

cd growPics/

# AVI
ffmpeg -pattern_type glob -i '*.png' -framerate 1 -c:v libx264 -crf 10 out.avi

# MP4
ffmpeg -pattern_type glob -i '*.png' -s 1296x972 -c:v libx264 -vf fps=10 -pix_fmt yuv420p out.mp4
